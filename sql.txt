CREATE DATABASE ChatServer;

CREATE TABLE users (
	Id INT NOT NULL AUTO_INCREMENT,
    Name VARCHAR(30) NOT NULL,
    LastName VARChar(30),
    Email VARCHAR(50),
    Pass VARCHAR(10),
  	Age int,
    PRIMARY KEY(id)
);
